\section{Preliminaries}
\label{sec:prelim}

\subsection{Additive Homomorphic Encryption}
Paillier cryptosystem provides additive homomorphic properties:
\begin{align}
    \text{Enc}_{pk}(x_1) \cdot \text{Enc}_{pk}(x_2) &= \text{Enc}_{pk}(x_1 + x_2) \\
    \text{Enc}_{pk}(x_1)^t &= \text{Enc}_{pk}(t \cdot x_1)
\end{align}

\subsection{Functional Inner Product}
For gradients viewed as functions over the input space:
\begin{equation}
    \langle g_i, g_{ref} \rangle = \int g_i(x) \, g_{ref}(x) \, dP(x) \approx \frac{1}{|B|} \sum_{x \in B} g_i(x) \, g_{ref}(x)
\end{equation}
The empirical approximation via batch averaging is exactly the gradient computation itself.

\section{FIP-FL: System Architecture and Algorithms}
\label{sec:algorithms}

We present three algorithms that define the complete FIP-FL protocol:
\begin{itemize}
    \item \textbf{Algorithm 1}: System initialization, key distribution, and reference bootstrapping
    \item \textbf{Algorithm 2}: Per-round federated learning with FIP verification (core protocol)
    \item \textbf{Algorithm 3}: Cryptographic primitives and utility functions
\end{itemize}

% ALGORITHM 1: SYSTEM SETUP
%==============================================================================
\begin{algorithm}[H]
\caption{FIP-FL System Setup and Key Distribution}
\label{alg:setup}
\begin{algorithmic}[1]
\Require Number of clients $N$, public bootstrap dataset $\mathcal{D}_{\text{pub}}$, security parameter $\lambda$
\Ensure Initialized FIP Verifier, global model $\mathbf{M}^{(0)}$, registered clients with key pairs

\Statex \textbf{Phase 1: Key Generation}
\State Trusted Key Authority (TKA) generates Paillier key pair:
\State \hspace{\algorithmicindent} $(pk, sk) \gets \text{KeyGen}(1^\lambda)$

\Statex \textbf{Phase 2: User Registration}
\For{$i \gets 1$ \textbf{to} $N$}
    \State Register user $i$, assign unique $\text{Tag\_ID}_i$
    \State Distribute public key $pk$ to user $i$
    \State Initialize local model: $\mathbf{W}_i \gets \mathbf{0}$, $\mathbf{b}_i \gets \mathbf{0}$
\EndFor

\Statex \textbf{Phase 3: Reference Bootstrapping}
\State Compute initial reference gradient on public data:
\State $\mathbf{R}^{(0)} \gets \nabla_{\bm{\theta}} \mathcal{L}(f_{\bm{\theta}}(\mathcal{D}_{\text{pub}}), y_{\text{pub}}) \big|_{\bm{\theta}=0}$

\Statex \textbf{Phase 4: System Initialization}
\State FIP Verifier initializes client history $\mathcal{H} \gets \emptyset$
\State FIP Verifier initializes update table $\mathcal{U} \gets \emptyset$
\State Aggregation Server initializes global model $\mathbf{M}^{(0)} = [\mathbf{W}^{(0)}; \mathbf{b}^{(0)}] = \mathbf{0}$

\Statex \textbf{Phase 5: Encryption Scheme Definition}
\Statex $\mathcal{E} = (\text{KeyGen}, \text{Enc}, \text{Dec})$ with additive homomorphic properties:
\Statex \hspace{\algorithmicindent} $\text{Enc}_{pk}(x_1) \cdot \text{Enc}_{pk}(x_2) = \text{Enc}_{pk}(x_1 + x_2)$
\Statex \hspace{\algorithmicindent} $\text{Enc}_{pk}(x_1)^t = \text{Enc}_{pk}(t \cdot x_1)$
\end{algorithmic}
\end{algorithm}

%==============================================================================
% ALGORITHM 2: PER-ROUND PROTOCOL (PART 1)
%==============================================================================
\begin{algorithm}[H]
\caption{FIP-FL Per-Round Verification Protocol (Part 1: Local Training)}
\label{alg:round1}
\begin{algorithmic}[1]
\Require Global model $\mathbf{M}^{(t)}$, reference $\mathbf{R}^{(t)}$, client datasets $\{\mathcal{D}_i\}_{i=1}^N$, round $t$
\Ensure Encrypted updates $[\![\Delta\bm{\theta}_i]\!]$ for all clients

\Statex \textbf{PHASE I: Local Training and Encrypted Update Generation}

\State $d \gets \text{input\_dim} \cdot \text{num\_classes}$ \Comment{Total dimension of flattened gradient}

\For{client $i = 1$ \textbf{to} $N$ \textbf{in parallel}}
    \State $\mathbf{y}_{\text{train}} \gets \mathbf{y}_i$
    
    \If{client $i$ is \textsc{Malicious}}
        \State $\mathbf{y}_{\text{train}} \gets (\mathbf{y}_{\text{train}} + 5) \bmod 10$ \Comment{Untargeted label flipping}
    \EndIf
    
    \State $\mathbf{W}_{\text{prev}} \gets \mathbf{W}_i$, $\mathbf{b}_{\text{prev}} \gets \mathbf{b}_i$
    
    \For{$\text{epoch} = 1$ \textbf{to} $E$}
        \For{each batch $\mathcal{B} \subseteq \mathcal{D}_i$}
            \State $\mathbf{X}_b, \mathbf{y}_b \gets \mathcal{B}$
            \State $\mathbf{s} \gets \mathbf{X}_b \mathbf{W}_i + \mathbf{b}_i$
            \State $\mathbf{p} \gets \text{softmax}(\mathbf{s})$
            \State $\mathbf{y}_{\text{oh}} \gets \text{one\_hot}(\mathbf{y}_b)$
            \State $\mathbf{e} \gets \mathbf{p} - \mathbf{y}_{\text{oh}}$
            \State $\nabla \mathbf{W} \gets (\mathbf{X}_b^\top \mathbf{e}) / |\mathcal{B}|$
            \State $\nabla \mathbf{b} \gets \text{mean}(\mathbf{e}, \text{axis}=0)$
            \State $\mathbf{W}_i \gets \mathbf{W}_i - \eta \nabla \mathbf{W}$
            \State $\mathbf{b}_i \gets \mathbf{b}_i - \eta \nabla \mathbf{b}$
        \EndFor
    \EndFor
    
    \State $\Delta \mathbf{W}_i \gets \mathbf{W}_{\text{prev}} - \mathbf{W}_i$
    \State $\Delta \mathbf{b}_i \gets \mathbf{b}_{\text{prev}} - \mathbf{b}_i$
    \State $\Delta \bm{\theta}_i \gets \text{flatten}([\Delta \mathbf{W}_i; \Delta \mathbf{b}_i])$
    \State $[\![\Delta \bm{\theta}_i]\!] \gets \text{Enc}_{pk}(\Delta \bm{\theta}_i \cdot \text{scale})$
    \State $\text{Send}(\text{Tag\_ID}_i, [\![\Delta \bm{\theta}_i]\!])$ to Update Table $\mathcal{U}$
\EndFor
\end{algorithmic}
\end{algorithm}

%==============================================================================
% ALGORITHM 2: PER-ROUND PROTOCOL (PART 2)
%==============================================================================
\begin{algorithm}[H]
\caption{FIP-FL Per-Round Verification Protocol (Part 2: FIP Verification)}
\label{alg:round2}
\begin{algorithmic}[1]
\Require Update table $\mathcal{U} = \{(\text{Tag\_ID}_i, [\![\Delta\bm{\theta}_i]\!])\}_{i=1}^N$, reference $\mathbf{R}^{(t)}$
\Ensure Set of accepted clients $\mathcal{A}$, optimized packet $\mathcal{P}$

\Statex \textbf{PHASE II: FIP Verification and Malicious Detection}

\For{each $(\text{Tag\_ID}_i, [\![\Delta \bm{\theta}_i]\!]) \in \mathcal{U}$}
    \State $[\![s_i]\!] \gets \sum_{j=1}^{d} [\![\Delta \bm{\theta}_{i,j}]\!] \cdot \text{encode}(\mathbf{R}^{(t)}_j)$ \Comment{$\text{encode}$ defined in Algorithm 3}
    \State $s_i \gets \text{Dec}_{sk}([\![s_i]\!]) / \text{scale}^2$
    \State $\text{Scores}[\text{Tag\_ID}_i] \gets s_i$
\EndFor

\State $\mathcal{S} \gets \text{values}(\text{Scores})$
\State $\mu \gets \text{median}(\mathcal{S})$
\State $\text{mad} \gets \text{median}(|\mathcal{S} - \mu|)$
\State $\text{mad} \gets \max(\text{mad}, \varepsilon)$

\State $n_{\text{neg}} \gets |\{s \in \mathcal{S} : s \leq 0\}|$
\State $\rho \gets n_{\text{neg}} / |\mathcal{S}|$ \Comment{Estimated attack rate}

\If{$t < T_{\text{warmup}}$}
    \State $\tau_{\min} \gets \max(\mu \cdot 0.1, 10^{-6})$
    \State $\tau_{\max} \gets \mu \cdot 20.0$
\Else
    \If{$\rho < 0.15$} $\tau_{\min} \gets \max(\mu \cdot 0.55, 10^{-6})$
    \ElsIf{$\rho < 0.25$} $\tau_{\min} \gets \max(\mu \cdot 0.50, 10^{-6})$
    \ElsIf{$\rho < 0.35$} $\tau_{\min} \gets \max(\mu \cdot 0.40, 10^{-6})$
    \Else $\tau_{\min} \gets \max(\mu \cdot 0.30, 10^{-6})$
    \EndIf
    \State $\tau_{\max} \gets \mu + (5.0 \cdot \text{mad})$
\EndIf

\State $\mathcal{A} \gets \emptyset$

\For{each $(\text{Tag\_ID}_i, s_i) \in \text{Scores}$}
    \State $\text{benign} \gets \text{True}$
    
    \If{$s_i \leq 0$} $\text{benign} \gets \text{False}$ \Comment{Poisoning attack}
    \ElsIf{$s_i < \tau_{\min}$} $\text{benign} \gets \text{False}$ \Comment{Scaling attack}
    \ElsIf{$s_i > \tau_{\max}$} $\text{benign} \gets \text{False}$ \Comment{Destabilization}
    \EndIf
    
    \If{$\text{benign}$ \textbf{and} $t \geq T_{\text{warmup}}$ \textbf{and} $\text{Tag\_ID}_i \in \mathcal{H}$}
        \State $\mu_i \gets \text{mean}(\mathcal{H}[\text{Tag\_ID}_i])$
        \State $\sigma_i \gets \max(\text{std}(\mathcal{H}[\text{Tag\_ID}_i]), 10^{-3})$
        \State $z_i \gets |s_i - \mu_i| / \sigma_i$
        \If{$z_i > Z_{\text{thresh}}$} $\text{benign} \gets \text{False}$ \Comment{Adaptive attack}
        \EndIf
    \EndIf
    
    \If{$\text{benign}$}
        \State $\mathcal{A} \gets \mathcal{A} \cup \{\text{Tag\_ID}_i\}$
        \If{$\text{Tag\_ID}_i \notin \mathcal{H}$} $\mathcal{H}[\text{Tag\_ID}_i] \gets []$ \EndIf
        \State $\mathcal{H}[\text{Tag\_ID}_i].\text{append}(s_i)$
        \If{$|\mathcal{H}[\text{Tag\_ID}_i]| > 10$} $\mathcal{H}[\text{Tag\_ID}_i].\text{pop}(0)$ \EndIf
    \EndIf
\EndFor

\State $\alpha \gets |\mathcal{A}| / |\mathcal{S}|$ \Comment{Acceptance ratio}
\end{algorithmic}
\end{algorithm}

%==============================================================================
% ALGORITHM 2: PER-ROUND PROTOCOL (PART 3)
%==============================================================================
\begin{algorithm}[H]
\caption{FIP-FL Per-Round Verification Protocol (Part 3: Crisis Recovery)}
\label{alg:round3}
\begin{algorithmic}[1]
\Require Scores, acceptance set $\mathcal{A}$, estimated attack rate $\rho$, round $t$
\Ensure Updated acceptance set $\mathcal{A}$ after crisis recovery

\If{$\alpha < 0.6$ \textbf{and} $t \geq T_{\text{warmup}}$}
    \If{$\rho > 0.25$}
        \State $\text{print}(\text{"[CRISIS] High attack rate detected. Recovering..."})$
    \Else
        \State $\text{print}(\text{"[WARNING] False positive cascade. Forcing acceptance..."})$
    \EndIf
    
    \State $\mathcal{A}_{\text{forced}} \gets \emptyset$
    \For{each $(\text{Tag\_ID}_i, s_i) \in \text{Scores}$}
        \If{$s_i > 0$}
            \State $\mathcal{A}_{\text{forced}} \gets \mathcal{A}_{\text{forced}} \cup \{\text{Tag\_ID}_i\}$
            \If{$\text{Tag\_ID}_i \notin \mathcal{H}$} $\mathcal{H}[\text{Tag\_ID}_i] \gets []$ \EndIf
            \State $\mathcal{H}[\text{Tag\_ID}_i].\text{append}(s_i)$
            \If{$|\mathcal{H}[\text{Tag\_ID}_i]| > 10$} $\mathcal{H}[\text{Tag\_ID}_i].\text{pop}(0)$ \EndIf
        \EndIf
    \EndFor
    \State $\mathcal{A} \gets \mathcal{A}_{\text{forced}}$
    \State $\alpha \gets |\mathcal{A}| / |\mathcal{S}|$
\EndIf

\State $\mathcal{P} \gets \emptyset$ \Comment{Optimized packet with deduplication}

\For{each $\text{Tag\_ID}_i \in \mathcal{A}$}
    \State $h \gets \text{hash}([\![\Delta \bm{\theta}_i]\!])$
    \If{$h \notin \mathcal{P}$}
        \State $\mathcal{P}[h] \gets \{\text{count}: 1, \text{enc\_data}: [\![\Delta \bm{\theta}_i]\!]\}$
    \Else
        \State $\mathcal{P}[h].\text{count} \gets \mathcal{P}[h].\text{count} + 1$
    \EndIf
\EndFor

\State $\text{Send}(\mathcal{P})$ to Aggregation Server
\State $\mathcal{U} \gets \emptyset$
\end{algorithmic}
\end{algorithm}

%==============================================================================
% ALGORITHM 2: PER-ROUND PROTOCOL (PART 4)
%==============================================================================
\begin{algorithm}[H]
\caption{FIP-FL Per-Round Verification Protocol (Part 4: Secure Aggregation)}
\label{alg:round4}
\begin{algorithmic}[1]
\Require Optimized packet $\mathcal{P}$ from Verifier
\Ensure Updated global model $\mathbf{M}^{(t+1)}$, aggregated gradient $\Delta_{\text{global}}$

\Statex \textbf{PHASE III: Secure Aggregation}

\If{$\mathcal{P} \neq \emptyset$}
    \State $[\![\Delta \bm{\Theta}_{\text{agg}}]\!] \gets 0$
    \State $\text{total} \gets 0$
    
    \For{each $(h, \text{packet}) \in \mathcal{P}$}
        \State $[\![\Delta \bm{\theta}]\!] \gets \text{packet.enc\_data}$
        \State $\text{count} \gets \text{packet.count}$
        \State $[\![\Delta \bm{\Theta}_{\text{agg}}]\!] \gets [\![\Delta \bm{\Theta}_{\text{agg}}]\!] + ([\![\Delta \bm{\theta}]\!] \cdot \text{count})$
        \State $\text{total} \gets \text{total} + \text{count}$
    \EndFor
    
    \State $\Delta \bm{\Theta}_{\text{flat}} \gets \text{Dec}_{sk}([\![\Delta \bm{\Theta}_{\text{agg}}]\!]) / \text{scale}$
    \State $\Delta \bm{\Theta}_{\text{avg}} \gets \Delta \bm{\Theta}_{\text{flat}} / \text{total}$
    
    \State $d_w \gets \text{input\_dim} \cdot \text{num\_classes}$
    \State $\Delta \mathbf{W} \gets \text{reshape}(\Delta \bm{\Theta}_{\text{avg}}[:d_w], (\text{input\_dim}, \text{num\_classes}))$
    \State $\Delta \mathbf{b} \gets \Delta \bm{\Theta}_{\text{avg}}[d_w:]$
    
    \State $\mathbf{W}^{(t+1)} \gets \mathbf{W}^{(t)} - \Delta \mathbf{W}$
    \State $\mathbf{b}^{(t+1)} \gets \mathbf{b}^{(t)} - \Delta \mathbf{b}$
    \State $\mathbf{M}^{(t+1)} \gets [\mathbf{W}^{(t+1)}; \mathbf{b}^{(t+1)}]$
    
    \State $\text{Send}(\Delta \bm{\Theta}_{\text{avg}})$ to FIP Verifier
\Else
    \State $\text{Send}(\text{NULL})$ to FIP Verifier
\EndIf
\end{algorithmic}
\end{algorithm}

%==============================================================================
% ALGORITHM 2: PER-ROUND PROTOCOL (PART 5)
%==============================================================================
\begin{algorithm}[H]
\caption{FIP-FL Per-Round Verification Protocol (Part 5: Reference Update)}
\label{alg:round5}
\begin{algorithmic}[1]
\Require Aggregated gradient $\Delta_{\text{global}}$, acceptance ratio $\alpha$
\Ensure Updated reference gradient $\mathbf{R}^{(t+1)}$

\Statex \textbf{PHASE IV: Reference Gradient Update}

\If{$\Delta_{\text{global}} \neq \text{NULL}$}
    \If{$\alpha > 0.7$}
        \State $\beta \gets 0.8$ \Comment{Normal momentum}
    \Else
        \State $\beta \gets 0.95$ \Comment{Conservative update during crisis}
    \EndIf
    
    \State $\mathbf{R}^{(t+1)} \gets \beta \cdot \mathbf{R}^{(t)} + (1 - \beta) \cdot \Delta_{\text{global}}$
    
    \If{$\text{mean}(\mathbf{R}^{(t+1)}) < 0$}
        \State $\mathbf{R}^{(t+1)} \gets -\mathbf{R}^{(t+1)}$
        \State $\text{print}(\text{"[WARNING] Reference gradient re-centered"})$
    \EndIf
\Else
    \State $\mathbf{R}^{(t+1)} \gets \mathbf{R}^{(t)}$
\EndIf

\Statex \textbf{PHASE V: Model Distribution}

\State Aggregation Server broadcasts $\mathbf{M}^{(t+1)}$ to all clients

\For{client $i = 1$ \textbf{to} $N$}
    \State Receive $\mathbf{M}^{(t+1)}$
    \State $\mathbf{W}_i \gets \mathbf{W}^{(t+1)}$
    \State $\mathbf{b}_i \gets \mathbf{b}^{(t+1)}$
\EndFor
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\caption{Cryptographic Primitives and Utility Functions}
\label{alg:utils}
\begin{algorithmic}[1]

\Function{HomomorphicInnerProduct}{$[\![\mathbf{g}]\!]$, $\mathbf{r}$}
    \State $\mathbf{r}_{\text{int}} \gets [\text{encode}(r_j) \text{ for } j = 1 \text{ to } d]$
    \State $[\![s]\!] \gets [\![g_1]\!] \cdot \mathbf{r}_{\text{int}}[1]$
    \For{$j \gets 2$ \textbf{to} $d$}
        \State $[\![s]\!] \gets [\![s]\!] + ([\![g_j]\!] \cdot \mathbf{r}_{\text{int}}[j])$
    \EndFor
    \State \Return $[\![s]\!]$
\EndFunction

\Function{Encode}{$x$}
    \State \Return $\text{round}(x \cdot \text{scale})$
\EndFunction

\Function{Decode}{$z$}
    \State \Return $z / \text{scale}$
\EndFunction

\Function{DecodeSquared}{$z$}
    \State \Return $z / \text{scale}^2$
\EndFunction

\Function{Softmax}{$\mathbf{z}$}
    \State $\mathbf{z}' \gets \mathbf{z} - \max(\mathbf{z}, \text{axis}=1)$
    \State $\mathbf{p} \gets \exp(\mathbf{z}') / \sum \exp(\mathbf{z}')$
    \State \Return $\mathbf{p}$
\EndFunction

\Function{OneHot}{$\mathbf{y}$, num\_classes $=10$}
    \State $\mathbf{Y} \gets \mathbf{0}_{|\mathbf{y}| \times \text{num\_classes}}$
    \For{$i \gets 1$ \textbf{to} $|\mathbf{y}|$}
        \State $\mathbf{Y}[i, \mathbf{y}[i]] \gets 1$
    \EndFor
    \State \Return $\mathbf{Y}$
\EndFunction

\Function{Flatten}{$\mathbf{W}$, $\mathbf{b}$}
    \State \Return $[\mathbf{W}[:]; \mathbf{b}]$
\EndFunction

\Function{Serialize}{$[\![\mathbf{g}]\!]$}
    \State \Return $\text{concatenate}([ \text{str}(c) \text{ for } c \text{ in } [\![\mathbf{g}]\!] ])$
\EndFunction

\Function{Hash}{$[\![\mathbf{g}]\!]$}
    \State \Return $\text{SHA256}(\text{Serialize}([\![\mathbf{g}]\!]))$
\EndFunction
\end{algorithmic}
\end{algorithm}

\section{Theoretical Analysis}
\label{sec:theory}

\begin{theorem}[Gradient Privacy]
\label{thm:privacy}
The FIP-FL verification mechanism reveals only $K = 1$ scalar value 
$s_i = \langle \mathbf{g}_i, \mathbf{R} \rangle$ per client per round.
For $d$-dimensional gradients with $d \gg 1$, recovering $\mathbf{g}_i$ from $s_i$ 
is information-theoretically impossible as the linear system
$\mathbf{R}^\top \mathbf{g}_i = s_i$ is underdetermined 
(1 equation, $d$ unknowns). The solution space is a $(d-1)$-dimensional 
affine subspace. The mutual information $I(\mathbf{g}_i; s_i)$ is bounded by
$H(s_i) \ll H(\mathbf{g}_i)$ for $d \gg 1$.
\end{theorem}

\begin{theorem}[Attack Resilience]
\label{thm:security}
Any malicious update $\mathbf{g}_{\text{adv}}$ accepted by FIP-FL must satisfy:
\begin{align}
    \langle \mathbf{g}_{\text{adv}}, \mathbf{R} \rangle &> 0 \tag{Direction constraint} \\
    \tau_{\min} \leq \langle \mathbf{g}_{\text{adv}}, \mathbf{R} \rangle &\leq \tau_{\max} \tag{Magnitude constraint} \\
    \frac{|\langle \mathbf{g}_{\text{adv}}, \mathbf{R} \rangle - \mu_i|}{\sigma_i} &\leq Z_{\text{thresh}} \tag{Consistency constraint}
\end{align}
where $\mu_i$ and $\sigma_i$ are the client's historical mean and standard deviation.
Direct poisoning attacks ($\theta \approx 180^\circ$) yield $\langle \mathbf{g}_{\text{adv}}, \mathbf{R} \rangle \leq 0 \rightarrow$ \textsc{Rejected}.
Scaling attacks ($\|\mathbf{g}_{\text{adv}}\| \approx 0$) yield $\langle \mathbf{g}_{\text{adv}}, \mathbf{R} \rangle < \tau_{\min} \rightarrow$ \textsc{Rejected}.
Destabilization attacks ($\|\mathbf{g}_{\text{adv}}\| \gg 1$) yield $\langle \mathbf{g}_{\text{adv}}, \mathbf{R} \rangle > \tau_{\max} \rightarrow$ \textsc{Rejected}.
\end{theorem}

\begin{theorem}[Convergence Guarantee]
\label{thm:convergence}
Assume the loss function $F$ is $L$-smooth. With momentum reference update
$\mathbf{R}^{(t+1)} = \beta \mathbf{R}^{(t)} + (1-\beta) \nabla F(\mathbf{w}^{(t)})$
and $\tau_{\min} > 0$, FIP-FL ensures $\langle \nabla F(\mathbf{w}^{(t)}), \tilde{\mathbf{g}}^{(t)} \rangle > 0$
for the aggregated gradient $\tilde{\mathbf{g}}^{(t)}$. For learning rate $\eta \leq 1/L$,
the global loss decreases monotonically:
\begin{equation}
    F(\mathbf{w}^{(t+1)}) \leq F(\mathbf{w}^{(t)}) - \frac{\eta}{2} \|\nabla F(\mathbf{w}^{(t)})\|^2
\end{equation}
\end{theorem}

\subsection{Threshold Selection via Chebyshev's Inequality}
\label{sec:threshold}

The choice of thresholds follows from Chebyshev's inequality:
\begin{equation}
    P\!\left(\,|X - \mu| \geq k\sigma\right) \leq \frac{1}{k^2}
\end{equation}
Setting $k = 5$ ensures that at least $96\%$ of honest clients fall within
$\mu \pm 5\sigma$. Since $\sigma \approx 1.4826 \cdot \mathrm{MAD}$, we set:
\begin{equation}
    \tau_{\max} = \mu + 5 \cdot \mathrm{MAD}
\end{equation}
For the lower bound, we use adaptive floors
$\alpha \in \{0.55,\, 0.50,\, 0.40,\, 0.30\}$ based on the estimated attack
rate $\rho$, which correspond to $20\times$ to $3\times$ safety margins below
the median.
